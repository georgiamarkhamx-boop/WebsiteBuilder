            // Store the experience level

        }
        
        // Security mistakes in office simulation
        function showMistakeTooltip(mistakeId) {

            // Toggle tooltip visibility
            if (tooltip.classList.contains('visible')) {
                tooltip.classList.remove('visible');
            } else {
                // Hide all tooltips first
                document.querySelectorAll('.mistake-tooltip').forEach(tip => {
                    tip.classList.remove('visible');
                });
                
                // Show this tooltip
                tooltip.classList.add('visible');
                
                // Add to found mistakes if not already found
                if (!userProfile.mistakesFound.includes(mistakeId)) {
                    userProfile.mistakesFound.push(mistakeId);
                    document.getElementById('mistakes-found').textContent = userProfile.mistakesFound.length;
                    
                    // Add badge

                    badge.textContent = getMistakeName(mistakeId);
                    document.getElementById('mistake-badges').appendChild(badge);
                    
                    // Update score
                    userProfile.score += 5;
                    updateScore();
                    
                    // Enable continue button if all mistakes found

        // Password strength checker
        function checkPasswordStrength() {
            const password = document.getElementById('password-test').value;
            const strengthBar = document.getElementById('password-strength-bar');
            const feedback = document.getElementById('password-feedback');
            
            if (!password) {
                strengthBar.className = 'password-strength-bar';
                strengthBar.style.width = '0%';
                feedback.textContent = 'Enter a password to see its strength';
                return;
            }
            
            // Simple password strength calculation
            let strength = 0;
            
            // Length check
            if (password.length >= 12) {
                strength += 25;
            } else if (password.length >= 8) {
                strength += 10;
            }
            
            // Complexity checks
            if (/[A-Z]/.test(password)) strength += 15; // Uppercase
            if (/[a-z]/.test(password)) strength += 10; // Lowercase
            if (/[0-9]/.test(password)) strength += 15; // Numbers
            if (/[^A-Za-z0-9]/.test(password)) strength += 20; // Special chars
            
            // Variety check
            const uniqueChars = new Set(password).size;
            strength += Math.min(15, uniqueChars * 2);
            
            // Update UI
            if (strength < 40) {
                strengthBar.className = 'password-strength-bar password-weak';
                feedback.textContent = 'Weak: This password could be easily cracked';
                feedback.className = 'text-sm mt-2 text-red-500';
            } else if (strength < 70) {
                strengthBar.className = 'password-strength-bar password-medium';
                feedback.textContent = 'Medium: This password provides some protection';
                feedback.className = 'text-sm mt-2 text-yellow-500';
            } else {
                strengthBar.className = 'password-strength-bar password-strong';
                feedback.textContent = 'Strong: This password provides good protection';
                feedback.className = 'text-sm mt-2 text-green-500';
            }
            
            strengthBar.style.width = Math.min(100, strength) + '%';
        }
        
        // Progress tracking

                                  (document.getElementById('screen-training').classList.contains('active') ? 1 : 0) + // Training

            const progressPercentage = Math.round((completedSteps / totalSteps) * 100);
            
            // Update progress bar
            document.getElementById('progress-bar').style.width = progressPercentage + '%';
            document.getElementById('progress-percentage').textContent = progressPercentage + '%';
            
            updateScore();
        }
        
        function updateScore() {
            document.getElementById('security-score').textContent = userProfile.score;
        }
